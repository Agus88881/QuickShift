# 1. Usamos la imagen del SDK de .NET 8 para compilar
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# 2. Copiamos los archivos de proyecto para restaurar dependencias
# (Asegurate de que los nombres de carpeta coincidan con los tuyos)
COPY ["QuickShift.Api.csproj", "./"]
RUN dotnet restore "QuickShift.Api.csproj"

# 3. Copiamos todo el resto del código y compilamos
COPY . .
RUN dotnet build "QuickShift.Api.csproj" -c Release -o /app/build

# 4. Publicamos la app (la dejamos lista para correr)
FROM build AS publish
RUN dotnet publish "QuickShift.Api.csproj" -c Release -o /app/publish

# 5. Usamos una imagen más liviana (solo el Runtime) para ejecutar
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "QuickShift.Api.dll"]
